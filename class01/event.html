jQuery Syntax Reference

Basic Syntax


// jQuery selector
$(selector).action()

// Document ready
$(document).ready(function() { /* code */ });
// or shorthand
$(function() { /* code */ });

Common Selectors


// Element, ID, Class
$('p')              // All <p> elements
$('#id')           // Element with id="id"
$('.class')        // Elements with class="class"

// Multiple selectors
$('p, #id, .class')

// Attribute selectors
$('[href]')        // Elements with href attribute
$('[href="value"]') // Elements with href="value"

Events


// Common events
.click()           // Click event
.dblclick()        // Double-click
.mouseenter()      // Mouse enter
.mouseleave()      // Mouse leave
.on('event', fn)   // Attach event handler
.off('event')      // Remove event handler

// Example
$('#btn').click(function() { alert('Clicked!'); });

DOM Manipulation


// Get/Set content
.text()            // Get text content
.text('new')       // Set text content
.html()            // Get HTML content
.html('new')       // Set HTML content
.val()             // Get input value
.val('new')        // Set input value

// Attributes
.attr('href')      // Get href attribute
.attr('href', 'url') // Set href attribute
.removeAttr('href') // Remove attribute

// Add elements
.append('content') // Append content
.prepend('content') // Prepend content
.after('content')  // Insert after
.before('content') // Insert before

Effects


// Show/Hide
.hide()            // Hide element
.show()            // Show element
.toggle()          // Toggle show/hide

// Fading
.fadeIn()          // Fade in
.fadeOut()         // Fade out
.fadeToggle()      // Toggle fade

// Sliding
.slideDown()       // Slide down
.slideUp()         // Slide up
.slideToggle()     // Toggle slide

// Animation
.animate({styles}, speed, callback)

AJAX


// Basic AJAX
$.ajax({
    url: 'file.php',
    method: 'POST',
    data: { key: 'value' },
    success: function(response) { /* handle success */ },
    error: function() { /* handle error */ }
});

// Shorthand methods
$.get('file.php', function(data) { /* handle data */ });
$.post('file.php', { key: 'value' }, function(data) { /* handle data */ });
